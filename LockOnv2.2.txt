-- ===== CẤU HÌNH (TÙY CHỈNH TẠI ĐÂY) =====
local CONFIG = {
    AimPart = "HumanoidRootPart",   
    MaxDistance = 4000,              
    LockKey = Enum.KeyCode.LeftControl, 
    IgnoreWalls = true,             
    TargetTeam = false,
    Smoothness = 5, -- ĐỂ 1 ĐỂ KHÓA CỨNG (HARDLOCK KHÔNG ĐỘ TRỄ)
}

-- ===== KHỞI TẠO TÀI NGUYÊN =====
local Players = game:GetService("Players")
local RunService = game:GetService("RunService")
local UserInputService = game:GetService("UserInputService")
local LocalPlayer = Players.LocalPlayer
local Camera = workspace.CurrentCamera

local AimlockEnabled = false
local CurrentTarget = nil

-- ===== TẠO GIAO DIỆN NÚT BẬT/TẮT (HÌNH TRÒN) =====
local ScreenGui = Instance.new("ScreenGui", (gethui and gethui()) or LocalPlayer:WaitForChild("PlayerGui"))

ScreenGui.ResetOnSpawn = false

local ToggleButton = Instance.new("ImageButton", ScreenGui)
ToggleButton.Size = UDim2.new(0, 70, 0, 70)

ToggleButton.BackgroundTransparency = 0.3
ToggleButton.Draggable = true
Instance.new("UICorner", ToggleButton).CornerRadius = UDim.new(1, 0)

local UIStroke = Instance.new("UIStroke", ToggleButton)
UIStroke.Color = Color3.fromRGB(255, 255, 255)
UIStroke.Thickness = 1.5

local LockIcon = Instance.new("ImageLabel", ToggleButton)
LockIcon.Size = UDim2.new(0, 40, 0, 40)
LockIcon.Position = UDim2.new(0.5, -20, 0.5, -20)
LockIcon.BackgroundTransparency = 1
LockIcon.Image = "rbxassetid://237305954"
LockIcon.ImageColor3 = Color3.fromRGB(255, 255, 255)

-- ===== TẠO CROSSHAIR CHẾ ĐỘ HITBOX =====
local CrosshairColor = Color3.fromRGB(0, 255, 255)
local TargetGui = Instance.new("BillboardGui")
TargetGui.Name = "AimlockCrosshair"
TargetGui.AlwaysOnTop = true
TargetGui.Size = UDim2.new(5, 0, 5, 0)
TargetGui.LightInfluence = 0

Container.Size = UDim2.new(1, 0, 1, 0)

    line.BackgroundColor3 = CrosshairColor
    line.BorderSizePixel = 0
    line.Size = UDim2.new(wScale, 0, hScale, 0)
    line.Position = pos
    line.AnchorPoint = Vector2.new(0.5, 0.5)
    line.Rotation = rot
end

makeScaleLine(0.25, 0.03, UDim2.new(0.15, 0, 0.15, 0), -45)

-- ===== HÀM KIỂM TRA MỤC TIÊU TỐT NHẤT =====
local function getBestTarget()
    local char = LocalPlayer.Character
    if not char or not char:FindFirstChild("HumanoidRootPart") then return nil end

    local root = char.HumanoidRootPart
    local bestTarget = nil
    local shortestDistance = math.huge
    local viewportCenter = Camera.ViewportSize / 2

    for _, model in ipairs(workspace:GetDescendants()) do
        if model:IsA("Model") and model ~= char then
            local hum = model:FindFirstChildOfClass("Humanoid")
            local targetPart = model:FindFirstChild(CONFIG.AimPart)
            
            if hum and hum.Health > 0 and targetPart and not model:FindFirstChildOfClass("ForceField") then
                if CONFIG.TargetTeam then
                    local player = Players:GetPlayerFromCharacter(model)
                    if player and player.Team == LocalPlayer.Team then continue end
                end

                local distToTarget = (root.Position - targetPart.Position).Magnitude
                if distToTarget <= CONFIG.MaxDistance then
                    
                    if not CONFIG.IgnoreWalls then
                        local rayParams = RaycastParams.new()
                        rayParams.FilterType = Enum.RaycastFilterType.Exclude
                        rayParams.FilterDescendantsInstances = {char, model}
                        
                        local rayResult = workspace:Raycast(Camera.CFrame.Position, (targetPart.Position - Camera.CFrame.Position).Unit * distToTarget, rayParams)
                        if rayResult then continue end 
                    end

                    local screenPos, onScreen = Camera:WorldToViewportPoint(targetPart.Position)
                    if onScreen then
                        local screenDist = (Vector2.new(screenPos.X, screenPos.Y) - viewportCenter).Magnitude
                        if screenDist < shortestDistance then
                            shortestDistance = screenDist
                            bestTarget = model
                        end
                    end
                end
            end
        end
    end
    return bestTarget
end

-- ===== CẬP NHẬT TRẠNG THÁI UI =====
local function updateUI()
    if AimlockEnabled then
        if CurrentTarget then
            UIStroke.Color = CrosshairColor
            LockIcon.ImageColor3 = CrosshairColor
        else
            UIStroke.Color = Color3.fromRGB(255, 255, 50)
            LockIcon.ImageColor3 = Color3.fromRGB(255, 255, 50)
        end
    else
        UIStroke.Color = Color3.fromRGB(255, 255, 255)
        LockIcon.ImageColor3 = Color3.fromRGB(255, 255, 255)
        TargetGui.Parent = nil
    end
end

-- ===== VÒNG LẶP KHÓA CAMERA (HARDLOCK - KHÔNG LỎNG LẺO) =====
RunService:BindToRenderStep("HTDrobloxHardLock", Enum.RenderPriority.Camera.Value + 200, function(dt)
    if not AimlockEnabled then return end

    -- Tìm mục tiêu mới nếu mục tiêu cũ chết/thoát
    local needNewTarget = false
    if not CurrentTarget or not CurrentTarget.Parent then
        needNewTarget = true
    else
        local hum = CurrentTarget:FindFirstChildOfClass("Humanoid")
        local part = CurrentTarget:FindFirstChild(CONFIG.AimPart)
        if not hum or hum.Health <= 0 or not part then
            needNewTarget = true
        end
    end

    if needNewTarget then
        CurrentTarget = getBestTarget()
        updateUI()
    end

    if CurrentTarget then
        local targetPart = CurrentTarget:FindFirstChild(CONFIG.AimPart)
        if targetPart then
            if TargetGui.Parent ~= targetPart then
                TargetGui.Parent = targetPart
            end

            -- Lấy vị trí camera hiện tại (giữ đồng bộ với nhân vật)
            local currentCamPos = Camera.CFrame.Position
            local targetCFrame = CFrame.lookAt(currentCamPos, targetPart.Position)

            if CONFIG.Smoothness >= 1 then
                -- Bám cứng 100%, phản hồi tức thì
                Camera.CFrame = targetCFrame
            else
                -- Chỉ dùng Lerp khi bạn cố tình chỉnh Smoothness thấp hơn 1
                local lerpFactor = math.clamp(CONFIG.Smoothness * (dt * 60), 0, 1)
                Camera.CFrame = Camera.CFrame:Lerp(targetCFrame, lerpFactor)
            end
        end
    end
end)

-- ===== CHỨC NĂNG BẬT/TẮT =====
local function toggleAimlock()
    AimlockEnabled = not AimlockEnabled
    if not AimlockEnabled then
        CurrentTarget = nil
    end
    updateUI()
end

ToggleButton.MouseButton1Click:Connect(toggleAimlock)

UserInputService.InputBegan:Connect(function(input, gameProcessed)
    if gameProcessed then return end
    if input.KeyCode == CONFIG.LockKey then
        toggleAimlock()
    end
end)
